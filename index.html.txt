        // Hashtag Generator
        function generateHashtags() {
            const keyword = document.getElementById('hashtagKeyword').value.toLowerCase().trim();
            const platform = document.getElementById('hashtagPlatform').value;
            
            if(!keyword) {
                alert('Please enter a keyword');
                return;
            }

            const prefixes = {
                tiktok: ['viral', 'trending', 'fyp', 'foryou', 'tiktok', 'viralvideo'],
                instagram: ['insta', 'instagram', 'instadaily', 'instagood', 'photooftheday', 'picoftheday'],
                facebook: ['fb', 'facebook', 'viral', 'trending', 'social', 'community']
            };

            const suffixes = ['love', 'life', 'style', 'daily', 'official', 'world', 'hub', 'central', 'daily', 'addict'];
            
            const generated = [];
            
            // Generate combinations
            prefixes[platform].forEach(pre => {
                generated.push(pre + keyword);
                generated.push(keyword + pre);
            });
            
            suffixes.forEach(suf => {
                generated.push(keyword + suf);
            });
            
            // Add generic popular tags
            const popular = ['viral', 'trending', 'explore', 'love', 'follow', 'like', 'share', 'followforfollowback'];
            generated.push(...popular);
            
            // Remove duplicates and shuffle
            const unique = [...new Set(generated)].sort(() => Math.random() - 0.5).slice(0, 20);
            
            const container = document.getElementById('hashtagList');
            container.innerHTML = unique.map((tag, idx) => 
                `<span class="px-4 py-2 bg-gradient-to-r from-primary/10 to-secondary/10 rounded-full text-sm font-medium border border-primary/20 hover:border-primary cursor-pointer transition-all hover:scale-105" style="animation-delay: ${idx * 0.05}s">#${tag}</span>`
            ).join('');
            
            document.getElementById('hashtagResults').classList.remove('hidden');
        }

        // Engagement Calculator
        function calculateEngagement(e) {
            e.preventDefault();
            
            const followers = parseInt(document.getElementById('calcFollowers').value);
            const likes = parseInt(document.getElementById('calcLikes').value);
            const comments = parseInt(document.getElementById('calcComments').value);
            
            const totalEngagement = likes + comments;
            const rate = ((totalEngagement / followers) * 100).toFixed(2);
            
            // Animate counter
            animateValue(document.getElementById('engagementRate'), 0, rate, 1000, '%');
            
            document.getElementById('totalEngagement').textContent = totalEngagement.toLocaleString();
            
            // Rating
            let rating, colorClass, tips;
            if(rate >= 6) {
                rating = 'Excellent';
                colorClass = 'bg-green-100 text-green-800 dark:bg-green-900 dark:text-green-200';
                tips = ['Keep up the great work!', 'Engage with your top fans', 'Collaborate with similar creators'];
            } else if(rate >= 3) {
                rating = 'Good';
                colorClass = 'bg-blue-100 text-blue-800 dark:bg-blue-900 dark:text-blue-200';
                tips = ['Post consistently', 'Use more call-to-actions', 'Respond to comments quickly'];
            } else if(rate >= 1) {
                rating = 'Average';
                colorClass = 'bg-yellow-100 text-yellow-800 dark:bg-yellow-900 dark:text-yellow-200';
                tips = ['Analyze your best performing content', 'Post at optimal times', 'Use trending hashtags'];
            } else {
                rating = 'Needs Improvement';
                colorClass = 'bg-red-100 text-red-800 dark:bg-red-900 dark:text-red-200';
                tips = ['Check content quality', 'Engage with other accounts', 'Run a giveaway', 'Analyze audience insights'];
            }
            
            const ratingEl = document.getElementById('performanceRating');
            ratingEl.textContent = rating;
            ratingEl.className = `font-bold px-3 py-1 rounded-full text-sm ${colorClass}`;
            
            document.getElementById('tipsList').innerHTML = tips.map(tip => `<li>• ${tip}</li>`).join('');
            
            document.getElementById('calcEmpty').classList.add('hidden');
            document.getElementById('calcResults').classList.remove('hidden');
        }

        // Counter Animation
        function animateValue(obj, start, end, duration, suffix = '') {
            let startTimestamp = null;
            const step = (timestamp) => {
                if (!startTimestamp) startTimestamp = timestamp;
                const progress = Math.min((timestamp - startTimestamp) / duration, 1);
                const value = Math.floor(progress * (end - start) + start);
                obj.innerHTML = (suffix === '%' ? (progress * (end - start) + start).toFixed(2) : value.toLocaleString()) + suffix;
                if (progress < 1) {
                    window.requestAnimationFrame(step);
                }
            };
            window.requestAnimationFrame(step);
        }

        // Intersection Observer for counters
        const observerOptions = {
            threshold: 0.5
        };

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if(entry.isIntersecting) {
                    const counters = entry.target.querySelectorAll('.counter');
                    counters.forEach(counter => {
                        const target = parseInt(counter.getAttribute('data-target'));
                        const isPercentage = counter.nextElementSibling && counter.nextElementSibling.textContent.includes('%');
                        animateValue(counter, 0, target, 2000, isPercentage ? '%' : '+');
                    });
                    observer.unobserve(entry.target);
                }
            });
        }, observerOptions);

        document.querySelectorAll('.counter').forEach(counter => {
            observer.observe(counter.parentElement.parentElement);
        });

                        entry.target.style.transform = 'translateY(0)';
                    }
                });
            }, { threshold: 0.1 });

            revealElements.forEach(el => revealObserver.observe(el));
        });
